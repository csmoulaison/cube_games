start:
    ... init ...
    for(core_count) {
        thread = create_thread
        if(i_am_child) {
            goto thread_idle
        }
    }
    goto start_frame

thread_idle:
    if(pixel_region_counter < pixel_regions_len) {
        atomic_increment(pixel_region_counter)
        my_region = pixel_region_counter - 1 or whatever
        goto render_region
    } else if(i_am_host) {
        goto end_frame
    }

render_region:
    ... render_pixels ...
    goto thread_idle

end_frame:
    ... swap_buffers, prepare_next_frame, etc ...
    ... falls through to ...
start_frame:
    pixel_region_counter = 0
    goto thread_idle
